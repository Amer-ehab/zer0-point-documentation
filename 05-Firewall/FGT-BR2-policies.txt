! ============================================================
! FortiGate BR2 — Firewall Policies
! ============================================================

! Zone configuration
config system zone
    edit "ZONE_BR2_LAN"
        set interface "BR2-VLAN210-CLIENT" "BR2-VLAN220-SERVERS" "BR2-VLAN230-IT" "BR2-VLAN240-MKT" "BR2-VLAN250-MGMT"
    next
end

! Internet access policies
config firewall policy
    ! Client Internet Access
    edit 1
        set name "BR2_Client_Internet"
        set srcintf "BR2-VLAN210-CLIENT"
        set dstintf "virtual-wan-link"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "ALL"
        set nat enable
        set logtraffic all
        set av-profile "BR2_AV_STANDARD"
        set webfilter-profile "BR2_WEB_STANDARD"
        set application-list "BR2_APP_STANDARD"
        set ips-sensor "BR2_IPS_STANDARD"
        set ssl-ssh-profile "certificate-inspection"
        set dnsfilter-profile "BR2_DNS_STANDARD"
        set file-filter-profile "BR2_FILE_FILTER"
    next
    ! Servers Policy
    edit 2
        set name "BR2_Servers_Policy"
