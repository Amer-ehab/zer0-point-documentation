! ============================================================
! SW-HQ-CORE-02 — Secondary Core Switch Configuration
! Zer0-Po!nT Enterprise Network
! Role: Standby HSRP Gateway, STP Root Secondary, Inter-VLAN Routing
! ============================================================

enable
configure terminal

! =========================
! 1) Basic Device Settings
! =========================
hostname SW-HQ-CORE-02
no ip domain-lookup
service password-encryption

! Enable Layer 3 routing
ip routing

! =========================
! 2) Spanning Tree
! =========================
spanning-tree mode rapid-pvst
! Make Core-02 the STP Root Secondary for all campus VLANs
spanning-tree vlan 10,20,30,40,50,60,61,70 root secondary

! =========================
! 3) VLAN Database
! =========================
vlan 10
 name MGMT
vlan 20
 name SERVERS
vlan 30
 name USERS
vlan 40
 name SOC
vlan 50
 name DMZ
vlan 60
 name WIFI
vlan 61
 name WIFI_GUEST
vlan 70
 name CCTV
vlan 98
 name TRANSIT_CORE02_TO_FGT_PORT4

! =========================
! 4) Trunk Links
! =========================

! --- Core Interconnect ---
interface Ethernet0/0
 description CORE-INTERCONNECT-to-CORE01
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,40,50,60,61,70
 no shutdown

! --- Uplink to ACC-MGMT ---
interface Ethernet0/1
 description TRUNK-to-ACC-MGMT
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 no shutdown

! --- Uplink to ACC-SERVERS ---
interface Ethernet0/2
 description TRUNK-to-ACC-SERVERS
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 20
 no shutdown

! --- Uplink to ACC-USERS ---
interface Ethernet0/3
 description TRUNK-to-ACC-USERS
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 30
 no shutdown

! --- Uplink to ACC-SOC ---
interface Ethernet1/0
 description TRUNK-to-ACC-SOC
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 40
 no shutdown

! --- Uplink to ACC-WIFI ---
interface Ethernet1/1
 description TRUNK-to-ACC-WIFI
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 60,61
 no shutdown

! --- Uplink to ACC-CCTV ---
interface Ethernet1/2
 description TRUNK-to-ACC-CCTV
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 70
 no shutdown

! --- Transit to FortiGate (port4) ---
interface range Ethernet1/3, Ethernet2/0
 description TO_FGT_HA_PORT4_TRANSIT
 switchport mode access
 switchport access vlan 98
 spanning-tree portfast
 no shutdown

! =========================
! 5) SVIs (Gateways) + HSRP
! =========================
! Core-02 IP = .3 | HSRP VIP = .1 | Lower Priority = Standby

interface Vlan10
 description SVI-MGMT
 ip address 10.1.10.3 255.255.255.0
 standby 10 ip 10.1.10.1
 standby 10 priority 100
 standby 10 preempt
 no shutdown

 description SVI-SERVERS
 ip address 10.1.20.3 255.255.255.0
 standby 20 ip 10.1.20.1
 standby 20 priority 100
 standby 20 preempt
 no shutdown

interface Vlan30
 description SVI-USERS
 ip address 10.1.30.3 255.255.255.0
 standby 30 ip 10.1.30.1
 standby 30 priority 100
 standby 30 preempt
 no shutdown

interface Vlan40
 description SVI-SOC
 ip address 10.1.40.3 255.255.255.0
 standby 40 ip 10.1.40.1

 standby 40 preempt
 no shutdown

! =========================

! =========================
