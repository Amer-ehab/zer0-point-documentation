! ============================================================
! SW-BR1-ACC-01 — Branch-01 Access Switch
! Zer0-Po!nT Enterprise Network
! Role: Layer-2 Access, User Connectivity
! ============================================================

enable
configure terminal

hostname SW-BR1-ACC-01
no ip domain-lookup

! VLANs
vlan 110
 name BR1-CLIENT
vlan 120
 name BR1-MGMT
vlan 130
 name BR1-IT
vlan 140
 name BR1-MKT
exit

spanning-tree mode rapid-pvst

! Uplink to Core-01
interface Ethernet0/0
 description UPLINK_TO_SW-BR1-CORE-01
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 110,120,130,140
 switchport nonegotiate
 spanning-tree portfast trunk
 no shutdown
exit

! Uplink to Core-02
interface Ethernet0/1
 description UPLINK_TO_SW-BR1-CORE-02
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 110,120,130,140
 switchport nonegotiate
 spanning-tree portfast trunk
 no shutdown
exit

! Access Ports
interface Ethernet0/2
 description CLIENT_USERS
 switchport mode access
 switchport access vlan 110
 spanning-tree portfast
 spanning-tree bpduguard enable
 no shutdown
exit

interface Ethernet0/3
 description MGMT_DEVICES
 switchport mode access
 switchport access vlan 120
 spanning-tree portfast
 spanning-tree bpduguard enable
 no shutdown
exit

interface Ethernet0/4
 description IT_USERS
 switchport mode access
 switchport access vlan 130
 spanning-tree portfast
 spanning-tree bpduguard enable
 no shutdown
exit

interface Ethernet0/5
 description MKT_USERS
 switchport mode access
 switchport access vlan 140
 spanning-tree portfast
 spanning-tree bpduguard enable
 no shutdown
exit

end
write memory
